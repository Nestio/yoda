#### Attributes

<Row>
  <Col>
    <Properties>
      <Property name="ID" type="integer">
        A unique identifier for the Lease Transaction within Funnel.
      </Property>
      <Property name="type" type="string">
        The type of Lease Transaction (e.g., "Application", "Renewal").
      </Property>
      <Property name="profile_type" type="string">
        The profile type associated with the transaction (e.g., "Standard").
      </Property>
      <Property name="community" type="object">
        Details of the community where the lease applies.
        <Properties>
          <Property name="id" type="integer">
            The community ID.
          </Property>
          <Property name="display_name" type="string">
            The display name of the community.
          </Property>
        </Properties>
      </Property>
      <Property name="status" type="string">
        The current status of the Lease Transaction (e.g., "Approved").
      </Property>
      <Property name="quote" type="object">
        Details of the lease quote including pricing and dates.
        <Properties>
          <Property name="id" type="integer">
            The quote ID.
          </Property>
          <Property name="price" type="string">
            The total price of the lease.
          </Property>
          <Property name="base_price" type="string">
            The base price of the lease.
          </Property>
          <Property name="lease_start" type="string">
            The start date of the lease.
          </Property>
          <Property name="lease_end" type="string">
            The end date of the lease.
          </Property>
          <Property name="lease_term" type="integer">
            The duration of the lease in months.
          </Property>
          <Property name="expiration" type="string">
            The expiration date and time of the quote.
          </Property>
          <Property name="total_monthly_fees" type="string">
            The total monthly fees associated with the lease.
          </Property>
          <Property name="total_move_in_fees" type="string">
            The total move-in fees associated with the lease.
          </Property>
        </Properties>
      </Property>
      <Property name="unit" type="object">
        Details of the leased unit including ID, price, and address.
        <Properties>
          <Property name="id" type="integer">
            The unit ID.
          </Property>
          <Property name="unit_number" type="string">
            The unit number.
          </Property>
          <Property name="price" type="number">
            The price of the unit.
          </Property>
          <Property name="address" type="string">
            The address of the unit.
          </Property>
        </Properties>
      </Property>
      <Property name="applicants" type="array">
        List of applicants involved in the transaction.
        <Properties>
          <Property name="id" type="integer">
            The applicant's ID.
          </Property>
          <Property name="full_name" type="string">
            The full name of the applicant.
          </Property>
          <Property name="role_name" type="string">
            The role of the applicant in the transaction (e.g., "Primary
            Applicant").
          </Property>
        </Properties>
      </Property>
      <Property name="invitees" type="array">
        List of invitees associated with the transaction.
        <Properties>
          <Property name="id" type="integer">
            The invitee's ID.
          </Property>
          <Property name="full_name" type="string">
            The full name of the invitee.
          </Property>
          <Property name="email" type="string">
            The email address of the invitee.
          </Property>
          <Property name="phone_number" type="string">
            The phone number of the invitee.
          </Property>
        </Properties>
      </Property>
      <Property name="security_deposit" type="number">
        The amount of security deposit for the lease.
      </Property>
      <Property name="fees_itemized" type="array">
        Breakdown of itemized fees associated with the transaction.
        <Properties>
          <Property name="id" type="integer">
            The fee item's ID.
          </Property>
          <Property name="name" type="string">
            The name of the fee.
          </Property>
          <Property name="fee_type" type="string">
            The type of fee (e.g., "Application Fee", "Holding Deposit").
          </Property>
          <Property name="amount" type="number">
            The amount of the fee.
          </Property>
          <Property name="paid" type="boolean">
            Indicates if the fee has been paid.
          </Property>
          <Property name="waived" type="boolean">
            Indicates if the fee has been waived.
          </Property>
          <Property name="charge_success" type="boolean">
            Indicates if the fee charge was successful.
          </Property>
          <Property name="is_handled_offline" type="boolean">
            Indicates if the fee was handled offline.
          </Property>
          <Property name="payer" type="string">
            The payer of the fee.
          </Property>
          <Property name="paid_for" type="string">
            The entity for which the fee was paid.
          </Property>
        </Properties>
      </Property>
      <Property name="guest_card" type="object">
        Guest card details associated with the transaction.
        <Properties>
          <Property name="client_id" type="integer">
            The client ID associated with the guest card.
          </Property>
          <Property name="realpage_pms_attributes" type="object">
            Attributes specific to RealPage PMS.
            <Properties>
              <Property name="guestcard_id" type="array">
                Array of guest card IDs.
              </Property>
              <Property name="resident_household_id" type="array">
                Array of resident household IDs.
              </Property>
            </Properties>
          </Property>
        </Properties>
      </Property>
      <Property name="executed_lease_document" type="array">
        List of executed lease documents (under consideration).
      </Property>
      <Property name="created_at" type="string">
        Date and time when the transaction was created.
      </Property>
      <Property name="updated_at" type="string">
        Date and time when the transaction was last updated.
      </Property>
    </Properties>
