      detailed information about the applicant, including personal details,

      object serves as a central entity in managing lease applications and
      facilitates the screening and approval process.

#### Attributes

<Row>
  <Col>
    <Properties>
      <Property name="ID" type="integer">
        A Unique Funnel Identifier for the applicant.
      </Property>
      <Property name="lease_transaction_id" type="integer">
        The ID of the Lease Transaction associated with the applicant.
      </Property>
      <Property name="role" type="string">
        The role of the applicant in the lease transaction (e.g., "Primary
        Applicant").
      </Property>
      <Property name="first_name" type="string">
        The first name of the applicant.
      </Property>
      <Property name="middle_name" type="string">
        The middle name of the applicant.
      </Property>
      <Property name="last_name" type="string">
        The last name of the applicant.
      </Property>
      <Property name="suffix" type="string">
        The suffix of the applicant's name (e.g., "Jr").
      </Property>
      <Property name="email" type="string">
        The email address of the applicant.
      </Property>
      <Property name="address_street" type="string">
        The street address of the applicant.
      </Property>
      <Property name="address_line_2" type="string">
        Additional line of address (e.g., apartment number).
      </Property>
      <Property name="address_city" type="string">
        The city of the applicant's address.
      </Property>
      <Property name="address_state" type="string">
        The state of the applicant's address.
      </Property>
      <Property name="address_postal_code" type="string">
        The postal code of the applicant's address.
      </Property>
      <Property name="screening_eviction_recommendation" type="string">
        Recommendation from eviction screening (e.g., "Accept").
      </Property>
      <Property name="screening_criminal_recommendation" type="string">
        Recommendation from criminal screening (e.g., "Accept").
      </Property>
      <Property name="identity_verification_status" type="string">
        The status of identity verification (e.g., "Manually Verified").
      </Property>
      <Property name="warnings_and_cautions" type="array">
        Array of warnings and cautions associated with the applicant.
        <Properties>
          <Property name="type" type="string">
            The type of warning or caution (e.g., "Warning").
          </Property>
          <Property name="source" type="string">
            The source of the warning or caution (e.g., "Transunion Report").
          </Property>
          <Property name="description" type="string">
            The description of the warning or caution.
          </Property>
        </Properties>
      </Property>
      <Property name="financial_data" type="object">
        Financial information of the applicant.
        <Properties>
          <Property name="reported_no_income_assets" type="boolean">
            Indicates if the applicant reported no income/assets.
          </Property>
          <Property name="income_total" type="string">
            Total reported income of the applicant.
          </Property>
          <Property name="asset_total" type="string">
            Total reported assets of the applicant.
          </Property>
        </Properties>
      </Property>
      <Property name="employer" type="object">
        Employer details of the applicant.
        <Properties>
          <Property name="name" type="string">
            The name of the employer.
          </Property>
          <Property name="address_street" type="string">
            The street address of the employer.
          </Property>
          <Property name="address_line_2" type="string">
            Additional line of employer's address (e.g., floor number).
          </Property>
          <Property name="address_city" type="string">
            The city of the employer's address.
          </Property>
          <Property name="address_state" type="string">
            The state of the employer's address.
          </Property>
          <Property name="address_postal_code" type="string">
            The postal code of the employer's address.
          </Property>
        </Properties>
      </Property>
    </Properties>
